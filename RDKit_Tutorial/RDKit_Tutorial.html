RDKit Tutorial¶

Installation¶

Currently RDKit only supports conda installation or built from the source, and there is no pip installation available. Conda installation is highly recommended. https://www.rdkit.org/docs/Install.html

conda install -c conda-forge rdkit
conda install -c conda-forge rdkit=2021.09.1

# function to get the similarity values
def get_similarities(smiles_list, base_mol_fp, fpsize = 1024, check_smiles = False):
    
    # initialize similarity metrics
    sim_cosine = []
    sim_dice = []
    sim_euclidean = []

    for idx, smiles in enumerate(smiles_list):
        m = Chem.MolFromSmiles(smiles)

        # get the fingerprints for the molecule to compare
        m_fp = np.array(AllChem.GetMorganFingerprintAsBitVect(m, 3, nBits=1024))

        # calculate basic components
        both_AB = np.sum(base_mol_fp*m_fp)
        sum_A = np.sum(base_mol_fp)
        sum_B = np.sum(m_fp)

        #calculate similarities
        sim_cosine.append(both_AB / np.sqrt(sum_A*sum_B))
        sim_dice.append(2*both_AB / (sum_A + sum_B))
        sim_euclidean.append(np.sqrt((fpsize-sum_A-sum_B+2*both_AB) / fpsize))
            
    return sim_cosine, sim_dice, sim_euclidean
